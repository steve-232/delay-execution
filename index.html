Delay Execution - example 1

		const example1 = document.querySelector('.example-1');
		const example1child = document.querySelector('.example-1-child');
	
		new DelayExecution({ 
			elem: example1child,
			func: () => {
				example1child.classList.add('active')
			},
			parent: example1
		});

Delay Execution - example 2

		const example2 = document.querySelector('.example-2');
		const example2child = document.querySelector('.example-2-child');

		new DelayExecution({ 
			elem: example2child,
			func: () => {
				example2child.classList.add('active')
			},
			parent: example2,
			direction: 'h',
			pixelDelay: 150
		});
